SELECT
	QRY_INGRESOS_TOTALES_PBI.date AS fecha, 
	DM_PERIODO.Periodo,
	DM_PERIODO.period,
	DM_PERIODO.Trimestre,
	DM_PERIODO.`AÃ±o`,
	QRY_BRANCH_OFFICES.branch_office, 
	QRY_BRANCH_OFFICES.`names` as supervisor, 
	QRY_IND_SSS.`value`, 
	QRY_INGRESOS_TOTALES_PBI.ticket_number, 
	QRY_INGRESOS_TOTALES_PBI.abonados, 
	QRY_INGRESOS_TOTALES_PBI.net_amount, 
	QRY_INGRESOS_TOTALES_PBI.transbank,
	
	(QRY_INGRESOS_TOTALES_PBI.net_amount + QRY_INGRESOS_TOTALES_PBI.transbank) AS Venta_Neta, 
	(QRY_INGRESOS_TOTALES_PBI.net_amount + QRY_INGRESOS_TOTALES_PBI.transbank + QRY_INGRESOS_TOTALES_PBI.abonados) AS Ingresos, 
	((QRY_INGRESOS_TOTALES_PBI.net_amount + QRY_INGRESOS_TOTALES_PBI.transbank) * (QRY_IND_SSS.`value`)) AS Venta_SSS, 
	((QRY_INGRESOS_TOTALES_PBI.net_amount + QRY_INGRESOS_TOTALES_PBI.transbank + QRY_INGRESOS_TOTALES_PBI.abonados) * (QRY_IND_SSS.`value`)) AS Ingresos_SSS
	
FROM
	QRY_INGRESOS_TOTALES_PBI
	LEFT JOIN QRY_BRANCH_OFFICES
	ON QRY_INGRESOS_TOTALES_PBI.branch_office_id = QRY_BRANCH_OFFICES.branch_office_id
	LEFT JOIN QRY_IND_SSS
	ON QRY_INGRESOS_TOTALES_PBI.clave = QRY_IND_SSS.clave
	LEFT JOIN DM_PERIODO
	ON QRY_INGRESOS_TOTALES_PBI.date = DM_PERIODO.Fecha
WHERE	
	QRY_BRANCH_OFFICES.status_id = 15 AND
	#MONTH(`QRY_INGRESOS_TOTALES_PBI`.`date`) = ((MONTH(curdate())-1)) AND
	YEAR(`QRY_INGRESOS_TOTALES_PBI`.`date`) = YEAR(curdate())
ORDER BY
	QRY_INGRESOS_TOTALES_PBI.date ASC
